Q1 — Conversion Funnel کامل
SELECT
  COUNT(DISTINCT c.user_id)                                              AS registered,
  COUNT(DISTINCT v.user_id)                                              AS visited,
  COUNT(DISTINCT co.user_id)                                             AS used_ai,
  COUNT(DISTINCT o.user_id)                                              AS purchased,
  COUNT(DISTINCT CASE WHEN rc.cnt >= 2 THEN rc.user_id END)             AS repeat_buyers,
  ROUND(COUNT(DISTINCT co.user_id) * 100.0 / COUNT(DISTINCT c.user_id),1)  AS ai_adoption_pct,
  ROUND(COUNT(DISTINCT o.user_id)  * 100.0 / COUNT(DISTINCT c.user_id),1)  AS overall_cvr
FROM customers c
LEFT JOIN visits v        ON c.user_id = v.user_id
LEFT JOIN consultation co ON c.user_id = co.user_id
LEFT JOIN orders o         ON c.user_id = o.user_id
LEFT JOIN (SELECT user_id, COUNT(*) cnt FROM orders GROUP BY user_id) rc
       ON c.user_id = rc.user_id;
Q2 — Cohort CVR (نرخ تبدیل هر موج ثبت‌نام)
SELECT
  strftime('%Y-%m', c.signup_date)                                           AS cohort,
  COUNT(DISTINCT c.user_id)                                                   AS cohort_size,
  COUNT(DISTINCT o.user_id)                                                   AS buyers,
  ROUND(COUNT(DISTINCT o.user_id) * 100.0 / COUNT(DISTINCT c.user_id), 1)  AS cvr_pct
FROM customers c
LEFT JOIN orders o ON c.user_id = o.user_id
GROUP BY cohort ORDER BY cohort;
Q3 — رفتار نشست: خریداران vs غیرخریداران + Correlation
SELECT
  CASE WHEN o.user_id IS NOT NULL THEN 'خریدار' ELSE 'غیرخریدار' END     AS segment,
  COUNT(DISTINCT u.user_id)                                                  AS users,
  ROUND(AVG(u.avg_dur), 2)                                                 AS avg_session_min,
  ROUND(AVG(u.avg_pv),  2)                                                 AS avg_page_views
FROM (
  SELECT user_id,
    AVG((julianday(end_time) - julianday(start_time)) * 1440) AS avg_dur,
    AVG(page_views)                                                AS avg_pv
  FROM visits GROUP BY user_id
) u
LEFT JOIN (SELECT DISTINCT user_id FROM orders) o ON u.user_id = o.user_id
GROUP BY segment;

Q5 — درآمد ماهانه + رشد MoM + نرخ بازگشت (واحد: تومان IRT)
-- واحد: تومان (IRT) فرضی — مبلغ = تعداد × ۴۳۷,۱۰۰ تومان (تأیید شده 100٪)
SELECT
  strftime('%Y-%m', t.transaction_date)                                 AS ماه,
  COUNT(*)                                                            AS تعداد_سفارش,
  SUM(CAST(REPLACE(REPLACE(t.amount,'$',''),',','') AS INTEGER))  AS درآمد_تومان,
  AVG(CAST(REPLACE(REPLACE(t.amount,'$',''),',','') AS INTEGER))  AS میانگین_سفارش,
  SUM(CASE WHEN o.status='Returned' THEN 1 ELSE 0 END)           AS تعداد_برگشتی,
  ROUND(SUM(CASE WHEN o.status='Returned' THEN 1 ELSE 0 END) * 100.0
    / COUNT(*), 1)                                                   AS نرخ_بازگشت
FROM transactions t JOIN orders o ON t.order_id = o.order_id
GROUP BY ماه ORDER BY ماه;
Q6 — LTV Segmentation (تومان IRT فرضی)
SELECT
  CASE
    WHEN cnt = 1 THEN 'تک‌خریدار'
    WHEN cnt = 2 THEN 'بازگشتی'
    ELSE              'وفادار (۳+)'
  END                             AS سگمنت,
  COUNT(*)                         AS تعداد_مشتری,
  ROUND(AVG(spend), 0)           AS میانگین_LTV_تومان,
  ROUND(COUNT(*) * 100.0 / (SELECT COUNT(*) FROM (SELECT user_id FROM orders GROUP BY user_id)), 1) AS سهم_درصد
FROM (
  SELECT o.user_id,
    COUNT(DISTINCT o.order_id) cnt,
    SUM(CAST(REPLACE(REPLACE(t.amount,'$',''),',','') AS INTEGER)) spend
  FROM orders o JOIN transactions t ON o.order_id = t.order_id
  GROUP BY o.user_id
) GROUP BY سگمنت;
Q7 — نرخ بازگشت بر اساس دسته‌بندی
SELECT
  cat.name                                                             AS دسته,
  COUNT(o.order_id)                                                   AS کل_سفارش,
  SUM(CASE WHEN o.status='Returned' THEN 1 ELSE 0 END)            AS برگشتی,
  ROUND(SUM(CASE WHEN o.status='Returned' THEN 1 ELSE 0 END) * 100.0
    / COUNT(*), 1)                                                   AS نرخ_بازگشت
FROM orders o
JOIN orderdetails od  ON o.order_id  = od.order_id
JOIN products p       ON od.product_id = p.product_id
JOIN categories cat   ON p.category_id = cat.category_id
GROUP BY cat.name ORDER BY نرخ_بازگشت DESC;

خط بنچ‌مارک صنعت ۱۰٪ را به عنوان مرز قابل‌قبول در نظر می‌گیریم. تمام ۷ ماه بالاتر از این مرز بوده‌اند — این یک الگوی سیستماتیک است، نه استثناء. بدترین ماه مارس با ۳۰.۳٪ بوده که احتمالاً به مشکل launch-time (توصیفات نادقیق محصول یا عدم آمادگی تیم پشتیبانی) مربوط است.

Scaling آن کانال می‌تواند ترکیب مشتری را بهبود دهد و نرخ بازگشت را به زیر ۱۵٪ برساند.

Q8 — Campaign Performance + CPA + ROAS (بودجه به تومان IRT)
SELECT
  campaign_id,
  CAST(REPLACE(REPLACE(budget,'$',''),',','') AS INTEGER)          AS بودجه_تومان,
  clicks, conversions,
  ROUND(conversions * 100.0 / NULLIF(clicks, 0), 2)            AS CTR,
  ROUND(budget_irt / NULLIF(conversions, 0), 0)                AS CPA_تومان,
  ROUND(avg_aov / (budget_irt/NULLIF(conversions,0)), 2)         AS ROAS_تخمینی,
  JULIANDAY(end_date) - JULIANDAY(start_date)                    AS روز
FROM campaigns,
  (SELECT AVG(CAST(REPLACE(REPLACE(amount,'$',''),',','') AS INTEGER)) avg_aov FROM transactions)
ORDER BY CPA_تومان ASC;
Q9 — What-If: اگر همه بودجه به مدل C7 می‌رفت
-- سناریو بهینه‌سازی بودجه (تومان IRT)
WITH best AS (
  SELECT MIN(CAST(REPLACE(REPLACE(budget,'$',''),',','') AS INTEGER) /
    NULLIF(conversions,0)) AS best_cpa_irt FROM campaigns
)
SELECT
  (SELECT SUM(CAST(REPLACE(REPLACE(budget,'$',''),',','') AS INTEGER)) FROM campaigns) AS کل_بودجه_تومان,
  best_cpa_irt                                                                        AS بهترین_CPA_تومان,
  ROUND(کل_بودجه / best_cpa_irt)                                                     AS تبدیل_بالقوه,
  200                                                                                 AS تبدیل_واقعی
FROM best;
