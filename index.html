Q1 — فانل کامل کاربر
-- نرخ تبدیل هر مرحله از فانل SELECT COUNT(DISTINCT c.user_id) AS total_customers, COUNT(DISTINCT v.user_id) AS unique_visitors, COUNT(DISTINCT co.user_id) AS ai_consultation_users, COUNT(DISTINCT o.user_id) AS ordering_customers, ROUND(COUNT(DISTINCT o.user_id) * 100.0 / COUNT(DISTINCT c.user_id), 2) AS conversion_rate_pct, ROUND(COUNT(DISTINCT co.user_id) * 100.0 / COUNT(DISTINCT v.user_id), 2) AS ai_adoption_rate_pct FROM customers c LEFT JOIN visits v ON c.user_id = v.user_id LEFT JOIN consultation co ON c.user_id = co.user_id LEFT JOIN orders o ON c.user_id = o.user_id;
Q2 — کیفیت نشست کاربران
SELECT strftime('%Y-%m', start_time) AS month, COUNT(*) AS total_sessions, COUNT(DISTINCT user_id) AS unique_users, ROUND(AVG(page_views), 1) AS avg_page_views, ROUND(AVG( (julianday(end_time) - julianday(start_time)) * 1440 ), 1) AS avg_duration_min, SUM(CASE WHEN page_views = 1 THEN 1 ELSE 0 END) AS bounce_sessions FROM visits GROUP BY month ORDER BY month;
Q3 — نرخ تبدیل مشاوره AI به خرید
SELECT COUNT(DISTINCT co.user_id) AS consulted_users, COUNT(DISTINCT o.user_id) AS consulted_and_ordered, ROUND(COUNT(DISTINCT o.user_id) * 100.0 / NULLIF(COUNT(DISTINCT co.user_id), 0), 1) AS post_consultation_cvr, AVG(co.consultation_duration) AS avg_duration, AVG(co.consultaion_msg_count) AS avg_messages FROM consultation co LEFT JOIN orders o ON co.user_id = o.user_id;

Q4 — درآمد و سفارش‌های ماهانه
SELECT strftime('%Y-%m', t.transaction_date) AS month, COUNT(t.transaction_id) AS num_orders, SUM(CAST(REPLACE(REPLACE(t.amount, '$',''), ',','') AS REAL)) AS total_revenue, AVG(CAST(REPLACE(REPLACE(t.amount, '$',''), ',','') AS REAL)) AS avg_order_value, SUM(CASE WHEN o.status='Returned' THEN 1 ELSE 0 END) AS returned_orders FROM transactions t JOIN orders o ON t.order_id = o.order_id GROUP BY month ORDER BY month;
Q5 — پرفروش‌ترین محصولات
SELECT p.name, c.name AS category, SUM(od.quantity) AS total_units_sold, p.stock_quantity AS remaining_stock, ROUND(SUM(od.quantity) * 100.0 / (SELECT SUM(quantity) FROM orderdetails), 1) AS sales_share_pct FROM orderdetails od JOIN products p ON od.product_id = p.product_id JOIN categories c ON p.category_id = c.category_id GROUP BY p.product_id ORDER BY total_units_sold DESC LIMIT 10;
Q6 — روش‌های پرداخت و سهم هر روش
SELECT payment_type, COUNT(*) AS transaction_count, ROUND(COUNT(*) * 100.0 / (SELECT COUNT(*) FROM transactions), 1) AS share_pct, SUM(CAST(REPLACE(REPLACE(amount, '$',''), ',','') AS REAL)) AS total_revenue FROM transactions GROUP BY payment_type ORDER BY transaction_count DESC;

Q7 — عملکرد کمپین‌های مارکتینگ
SELECT campaign_id, start_date, end_date, CAST(REPLACE(REPLACE(budget, '$',''), ',','') AS REAL) AS budget_usd, clicks, conversions, ROUND(conversions * 100.0 / NULLIF(clicks, 0), 2) AS conversion_rate_pct, ROUND(budget_usd / NULLIF(conversions, 0), 0) AS cost_per_acquisition, JULIANDAY(end_date) - JULIANDAY(start_date) AS campaign_days FROM campaigns ORDER BY conversion_rate_pct DESC;
Q8 — ROAS و نرخ بازگشت سرمایه مارکتینگ
SELECT SUM(budget_usd) AS total_marketing_spend, SUM(conversions) AS total_conversions, ROUND(total_revenue / total_marketing_spend, 2) AS ROAS, ROUND(total_marketing_spend / total_conversions, 0) AS overall_CPA FROM ( SELECT SUM(CAST(REPLACE(REPLACE(budget, '$',''), ',','') AS REAL)) AS budget_usd, SUM(conversions) AS total_conversions, (SELECT SUM(CAST(REPLACE(REPLACE(amount, '$',''), ',','') AS REAL)) FROM transactions) AS total_revenue FROM campaigns ) sub;

مشکلات کشف‌شده — ۳ مشکل اصلی کسب‌وکار
مبتنی بر داده و مقایسه با بنچ‌مارک صنعت e-commerce آرایشی
P1
نرخ بازگشت کالا غیرعادی — ۲۲.۵٪
بنچ‌مارک صنعت: ۵–۱۰٪ | این عدد ۲ تا ۴ برابر نرم
🔴 بحرانی
شواهد از داده
  • از ۲۰۰ سفارش، ۴۵ سفارش برگشت خورده (۲۲.۵٪) در مقابل بنچ‌مارک ۵–۱۰٪
  • نرخ بازگشت در ماه مارس به اوج ۳۰٪ (۱۰ از ۳۳) رسیده
  • سیستم مشاوره AI فقط ۲.۶٪ کاربران استفاده می‌کنند — مزیت رقابتی اصلی غیرفعال است
  • نرخ تبدیل مشاوره→خرید فقط ۷.۷٪ — نشانه عدم تطابق محصول با انتظار
تحلیل ریشه‌ای
در e-commerce آرایشی، بالاترین دلیل برگشت کالا «عدم تطابق رنگ/پوست با توضیحات» است. استارتاپ یک سیستم AI مشاوره دارد که می‌توانست این مشکل را حل کند، اما تنها ۲.۶٪ کاربران از آن استفاده می‌کنند. این یعنی مزیت رقابتی اصلی محصول کاملاً نادیده گرفته شده و نرخ بازگشت را بالا نگه داشته. علاوه بر این، توضیحات محصول احتمالاً ناکافی بوده و تصاویر با واقعیت فاصله دارند.
P2
AI مشاوره بی‌استفاده — نرخ پذیرش ۲.۶٪
مزیت رقابتی اصلی محصول عملاً غیرفعال است
🟡 بالا
شواهد از داده
  • ۱۰۰٪ کاربران ثبت‌نام‌شده از سایت بازدید کرده‌اند — مشکل دسترسی نیست
  • تنها ۲۶ نفر از ۱۰۰۰ کاربر (۲.۶٪) از AI مشاوره استفاده کرده‌اند
  • از این ۲۶ نفر، فقط ۲ نفر (۷.۷٪) بعد از مشاوره خرید کرده‌اند
  • میانگین مدت نشست سایت ۸.۹ دقیقه است — کاربران وقت دارند اما AI را پیدا نمی‌کنند
تحلیل ریشه‌ای
دو مشکل موازی: اول، discoverability — کاربران AI مشاوره را پیدا نمی‌کنند یا نمی‌دانند وجود دارد. دوم، UX Flow — حتی آن ۲۶ نفر که مشاوره گرفتند، تنها ۷.۷٪ به خرید رسیدند که نشان می‌دهد transition از مشاوره به خرید شکسته است. در مقایسه با رقبا مثل Sephora که AI Color IQ را در صفحه اصلی برجسته می‌کند، این استارتاپ ویژگی اصلی خود را پنهان کرده.
P3
افت شدید درآمد از آوریل به اگوست — منهای ۴۴٪
روند نزولی پایدار در نیمه دوم دوره بررسی
🟡 بالا
شواهد از داده
  • درآمد از $32.7M در آوریل به $18.3M در اگوست — کاهش ۴۴٪ در ۴ ماه
  • تعداد سفارش از ۳۵ در آوریل به ۲۴ در اگوست — کاهش ۳۱٪
  • نرخ خرید مجدد فقط ۱۵.۶٪ — اکثر مشتریان بار اول و آخر خرید می‌کنند
  • هزینه کمپین‌ها $41.5M برای ۲۰۰ سفارش — CPA بالا بدون retention strategy
تحلیل ریشه‌ای
کسب‌وکار به جای ساختن LTV (Lifetime Value)، در یک حلقه جذب مشتری جدید گیر کرده. با نرخ خرید مجدد ۱۵.۶٪، هر مشتری جدید هزینه‌ای بالا دارد ($207K CPA) و بیشتر نمی‌مانند. این ترکیب با نرخ بازگشت بالا، یک چرخه سوزاندن بودجه بدون رشد پایدار ایجاد می‌کند. بنچ‌مارک e-commerce beauty: نرخ بازگشت مشتری ۳۰–۴۵٪ — این کسب‌وکار فقط ۱۵.۶٪ دارد.
